CREATE DATABASE IF NOT EXISTS wcms;
USE wcms;

-- =========================================================
-- THIRD NORMAL FORM (3NF) SCHEMA REFERENCE
-- =========================================================
-- This file documents a 3NF target schema and equivalent DAO SQL.
-- Existing Java DAOs would need matching SQL changes before using
-- this schema directly in the application.
-- This version removes repeated descriptive values and stores
-- relationships through lookup tables and foreign keys.

-- LOOKUP TABLES
CREATE TABLE account_role (
    role_id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(20) NOT NULL UNIQUE
);

CREATE TABLE collection_day (
    collection_day_id INT AUTO_INCREMENT PRIMARY KEY,
    day_name VARCHAR(15) NOT NULL UNIQUE
);

CREATE TABLE personnel_role (
    personnel_role_id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(30) NOT NULL UNIQUE
);

CREATE TABLE truck_type (
    truck_type_id INT AUTO_INCREMENT PRIMARY KEY,
    type_name VARCHAR(50) NOT NULL UNIQUE
);

-- ACCOUNT / AUTH
CREATE TABLE account (
    account_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email_address VARCHAR(120) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    age INT NULL,
    gender ENUM('Male', 'Female', 'Other') NULL,
    last_login DATETIME NULL,
    status ENUM('Active', 'Inactive', 'Deactivated') DEFAULT 'Active',
    role_id INT NOT NULL,

    FOREIGN KEY (role_id) REFERENCES account_role(role_id)
        ON UPDATE CASCADE
);

-- BARANGAY
CREATE TABLE barangay (
    barangay_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_name VARCHAR(100) NOT NULL UNIQUE,
    barangay_household INT DEFAULT 0,
    contact_number VARCHAR(50),
    collection_day_id INT NOT NULL,
    status ENUM('Sheduled', 'Not Scheduled',) DEFAULT 'Not Scheduled',

    FOREIGN KEY (collection_day_id) REFERENCES collection_day(collection_day_id)
        ON UPDATE CASCADE
);

-- BARANGAY ADMIN ASSIGNMENT
-- Personal details stay in account; this table only stores the assignment.
CREATE TABLE barangay_admin (
    barangay_admin_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_id INT NOT NULL UNIQUE,
    account_id INT NOT NULL UNIQUE,

    FOREIGN KEY (barangay_id) REFERENCES barangay(barangay_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (account_id) REFERENCES account(account_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

-- PERSONNEL
CREATE TABLE personnel (
    personnel_id INT AUTO_INCREMENT PRIMARY KEY,
    personnel_name VARCHAR(100) NOT NULL,
    age INT,
    gender VARCHAR(20),
    address VARCHAR(100),
    contact_number VARCHAR(20),
    personnel_role_id INT NOT NULL,
    status ENUM('Active', 'Inactive') DEFAULT 'Active',

    FOREIGN KEY (personnel_role_id) REFERENCES personnel_role(personnel_role_id)
        ON UPDATE CASCADE
);

-- TRUCK
CREATE TABLE truck (
    truck_id INT AUTO_INCREMENT PRIMARY KEY,
    plate_number VARCHAR(20) NOT NULL UNIQUE,
    truck_type_id INT NOT NULL,
    truck_image BLOB,
    status ENUM('Active', 'Maintenance', 'Inactive') DEFAULT 'Active',

    FOREIGN KEY (truck_type_id) REFERENCES truck_type(truck_type_id)
        ON UPDATE CASCADE
);

-- TEAM
CREATE TABLE team (
    team_id INT AUTO_INCREMENT PRIMARY KEY,
    team_name VARCHAR(50) NOT NULL UNIQUE,
    leader_id INT NOT NULL UNIQUE,
    driver_id INT NULL UNIQUE,
    truck_id INT NULL UNIQUE,
    status VARCHAR(30) NOT NULL,

    FOREIGN KEY (leader_id) REFERENCES personnel(personnel_id)
        ON UPDATE CASCADE,
    FOREIGN KEY (driver_id) REFERENCES personnel(personnel_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    FOREIGN KEY (truck_id) REFERENCES truck(truck_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- TEAM COLLECTOR ASSIGNMENT
CREATE TABLE team_collectors (
    team_id INT NOT NULL,
    personnel_id INT NOT NULL,
    PRIMARY KEY (team_id, personnel_id),
    UNIQUE KEY uq_team_collectors_personnel (personnel_id),

    FOREIGN KEY (team_id) REFERENCES team(team_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (personnel_id) REFERENCES personnel(personnel_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

-- SCHEDULE
CREATE TABLE schedule (
    schedule_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_id INT NOT NULL,
    team_id INT NULL,
    schedule_date DATE NOT NULL,
    schedule_time TIME NULL,
    status ENUM('Completed', 'Pending', 'Missed') DEFAULT 'Pending',

    UNIQUE KEY uq_schedule_barangay (barangay_id),
    FOREIGN KEY (barangay_id) REFERENCES barangay(barangay_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (team_id) REFERENCES team(team_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- COMPLAINT
CREATE TABLE complaint (
    complaint_id INT AUTO_INCREMENT PRIMARY KEY,
    message TEXT,
    proof BLOB,
    status ENUM('Approved', 'Under Review', 'Rejected'),
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    location VARCHAR(255)
);

-- REPORT
CREATE TABLE report (
    report_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_id INT,
    message TEXT,
    proof BLOB,
    status ENUM('Approved', 'Under Review', 'Rejected'),
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (barangay_id) REFERENCES barangay(barangay_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- REQUEST
CREATE TABLE request (
    request_id INT AUTO_INCREMENT PRIMARY KEY,
    message TEXT,
    proof BLOB,
    status ENUM('Approved', 'Under Review', 'Rejected'),
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    location VARCHAR(255)
);

CREATE DATABASE IF NOT EXISTS wcms;
USE wcms;

-- =========================================================
-- 1. AUTH / IDENTITY LAYER
-- =========================================================

CREATE TABLE accounts (
    account_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(120) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    last_login TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE roles (
    role_id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE account_roles (
    account_id BIGINT NOT NULL,
    role_id INT NOT NULL,
    PRIMARY KEY (account_id, role_id),
    FOREIGN KEY (account_id) REFERENCES accounts(account_id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES roles(role_id) ON DELETE CASCADE
);

-- =========================================================
-- 2. BARANGAY MODULE
-- =========================================================

CREATE TABLE collection_days (
    collection_day_id INT AUTO_INCREMENT PRIMARY KEY,
    day_name VARCHAR(20) NOT NULL UNIQUE
);

CREATE TABLE barangays (
    barangay_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(120) NOT NULL UNIQUE,
    household_count INT DEFAULT 0,
    contact_number VARCHAR(50),
    collection_day_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (collection_day_id) REFERENCES collection_days(collection_day_id)
);

CREATE TABLE barangay_admins (
    barangay_admin_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    barangay_id BIGINT NOT NULL UNIQUE,
    account_id BIGINT NOT NULL UNIQUE,
    FOREIGN KEY (barangay_id) REFERENCES barangays(barangay_id) ON DELETE CASCADE,
    FOREIGN KEY (account_id) REFERENCES accounts(account_id) ON DELETE CASCADE
);

);

CREATE TABLE personnel (
    personnel_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(120) NOT NULL,
    age INT,
    gender_id INT,
    address VARCHAR(255),
    contact_number VARCHAR(50),
    role_id INT NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (role_id) REFERENCES personnel_roles(role_id),
    FOREIGN KEY (gender_id) REFERENCES genders(gender_id)
);

-- =========================================================
-- 4. TRANSPORT MODULE
-- =========================================================

CREATE TABLE truck_types (
    truck_type_id INT AUTO_INCREMENT PRIMARY KEY,
    type_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE statuses (
    status_id INT AUTO_INCREMENT PRIMARY KEY,
    status_type VARCHAR(50) NOT NULL,
    status_name VARCHAR(50) NOT NULL,
    UNIQUE(status_type, status_name)
);

CREATE TABLE trucks (
    truck_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    plate_number VARCHAR(30) NOT NULL UNIQUE,
    truck_type_id INT NOT NULL,
    status_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (truck_type_id) REFERENCES truck_types(truck_type_id),
    FOREIGN KEY (status_id) REFERENCES statuses(status_id)
);

-- =========================================================
-- 5. TEAM SYSTEM (FULLY FLEXIBLE)
-- =========================================================

CREATE TABLE teams (
    team_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    team_name VARCHAR(80) NOT NULL UNIQUE,
    status_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (status_id) REFERENCES statuses(status_id)
);

CREATE TABLE team_member_roles (
    role_id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE team_members (
    team_id BIGINT NOT NULL,
    personnel_id BIGINT NOT NULL,
    member_role_id INT NOT NULL,

    PRIMARY KEY (team_id, personnel_id),

    FOREIGN KEY (team_id) REFERENCES teams(team_id) ON DELETE CASCADE,
    FOREIGN KEY (personnel_id) REFERENCES personnel(personnel_id) ON DELETE CASCADE,
    FOREIGN KEY (member_role_id) REFERENCES team_member_roles(role_id)
);

-- =========================================================
-- 6. SCHEDULING SYSTEM
-- =========================================================

CREATE TABLE schedules (
    schedule_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    barangay_id BIGINT NOT NULL,
    team_id BIGINT,
    schedule_datetime DATETIME NOT NULL,
    status_id INT NOT NULL,

    UNIQUE (barangay_id, schedule_datetime),

    FOREIGN KEY (barangay_id) REFERENCES barangays(barangay_id),
    FOREIGN KEY (team_id) REFERENCES teams(team_id),
    FOREIGN KEY (status_id) REFERENCES statuses(status_id)
);

-- =========================================================
-- 7. UNIFIED TICKETING SYSTEM
-- (replaces complaint, report, request)
-- =========================================================

CREATE TABLE tickets (
    ticket_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    ticket_type ENUM('COMPLAINT','REPORT','REQUEST') NOT NULL,
    message TEXT,
    proof BLOB,
    location VARCHAR(255),
    is_read BOOLEAN DEFAULT FALSE,
    status_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (status_id) REFERENCES statuses(status_id)
);
